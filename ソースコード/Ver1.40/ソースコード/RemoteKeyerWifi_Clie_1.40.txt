int rttToDelayMs(long rttMs)
{
  if (rttMs <= RTT_THRESH_1) return DELAY_VERY_LOW;
  if (rttMs <= RTT_THRESH_2) return DELAY_LOW;
  if (rttMs <= RTT_THRESH_3) return DELAY_MID;
  if (rttMs <= RTT_THRESH_4) return DELAY_HIGH;
  if (rttMs <= RTT_THRESH_5) return DELAY_VERY_HIGH;
  if (rttMs <= RTT_THRESH_6) return DELAY_VERY2_HIGH;
  if (rttMs <= RTT_THRESH_7) return DELAY_VERY3_HIGH;
  return DELAY_MAX;
}

void freeWiFiProfileStrings(WiFiProfile &p)
{
  if (p.ssid) { free(const_cast<char *>(p.ssid)); p.ssid = nullptr; }
  if (p.password) { free(const_cast<char *>(p.password)); p.password = nullptr; }
  if (p.hostname) { free(const_cast<char *>(p.hostname)); p.hostname = nullptr; }
  if (p.serverhostname) { free(const_cast<char *>(p.serverhostname)); p.serverhostname = nullptr; }
}

long getAverageRtt()
{
  long sum = 0;
  for (int i = 0; i < RTT_HISTORY_SIZE; i++)
    sum += rttHistory[i];
  return sum / RTT_HISTORY_SIZE;
}

void keyISR(void *arg)
{
  if (!usbMode && !timeSynced)
    return;
  unsigned long now = micros();
  if (now - lastIsrMicros < DEBOUNCE_US)
    return;
  lastIsrMicros = now;
  keyState = (digitalRead(KEY_PIN) == LOW);

  lastKeyChangeMicros = now;
  immediateSendRequested = true;
}
// ==== 初期化 ====
void setup()
{
  Serial.begin(115200);
  EEPROM.begin(EEPROM_SIZE);
#ifdef LoadFromSource
  profileCount = sizeof(defaultProfiles) / sizeof(WiFiProfile);
  for (int i = 0; i < profileCount && i < MAX_WIFI_PROFILES; i++)
  {
    wifiProfiles[i] = defaultProfiles[i];
  }
#endif
  auto cfg = M5.config();
  cfg.external_spk = false;  // 外部スピーカー無効
  cfg.internal_mic = false;  // 内蔵マイク無効
  cfg.output_power = false;  // 電源制御ピン無効
  cfg.clear_display = false; // 画面初期化あり（必要に応じて）
  M5.begin(cfg);

#ifdef LoadFromSource
#else
  bool sdAvailable = false;
  delay(500);
  SPI.begin(SD_SPI_SCK_PIN, SD_SPI_MISO_PIN, SD_SPI_MOSI_PIN, SD_SPI_CS_PIN);

  if (SD.begin(SD_SPI_CS_PIN) && SD.exists(SD_Filename))
  {
    // Serial.println("SDカード初期化成功！");
    if (loadWiFiProfilesFromSD())
    {
      clearEEPROM();
      for (int i = 0; i < profileCount; i++)
      {
        saveWiFiProfileToEEPROM(wifiProfiles[i], i);
      }
      selectedProfileIndex = 0;
      saveProfileIndexToEEPROM(0);

      SystemSettingsEEPROM sysSettings = {
          0xA5,
          keyRepeatCount,
          SAMPLE_SEND_DUP_CNT,
          SAMPLE_INTERVAL_MS,
          SAMPLE_SEND_AMNT,
          DEBOUNCE_US};
      saveSystemSettingsToEEPROM(sysSettings);

      delay(100);

      // ✅ 成功通知：青ランプ点灯
      initLed();
      flashLedProfile(&BlueProfile, 2000, 1);

      while (true)
        delay(100);
    }
    else
    {
      // ❌️ 失敗通知：赤ランプ点灯
      initLed();
      flashLedProfile(&RedProfile, 2000, 1);

      while (true)
        delay(100);
    }
  }
  else
  {
    // Serial.println("SDカードが見つからないか、設定ファイルがありません。EEPROMから復元します。");
  }

  // ✅ EEPROMからプロファイルを復元
  if (!restoreProfilesFromEEPROM())
  {
    initLed();
    while (true)
    {
      flashOnce(&RedProfile, BLINK_INTERVAL_SYNC);
      delay(BLINK_INTERVAL_SYNC);
    }
  }

  if (selectedProfileIndex >= profileCount)
  {
    // Serial.println("⚠️ EEPROMのプロファイルインデックスが無効です。0にリセットします");
    selectedProfileIndex = 0;
  }

  SystemSettingsEEPROM sysSettings;
  if (loadSystemSettingsFromEEPROM(sysSettings))
  {
    keyRepeatCount = constrain(sysSettings.keyRepeatCount, 1, 50);
    SAMPLE_SEND_DUP_CNT = constrain(sysSettings.sampleSendDupCnt, 1, 10);
    SAMPLE_INTERVAL_MS = constrain(sysSettings.sampleIntervalMs, 1, 100);
    SAMPLE_SEND_AMNT = constrain(sysSettings.sampleSendAmnt, 1, 16);
    DEBOUNCE_US = constrain(sysSettings.debounceUs, 100, 50000);
  }
  else
  {
    // デフォルト値を設定して保存
    sysSettings = {
        0xA5,
        10,  // keyRepeatCount
        1,   // sampleSendDupCnt
        10,  // sampleIntervalMs
        8,   // sampleSendAmnt
        5000 // debounceUs
    };
    saveSystemSettingsToEEPROM(sysSettings);
  }

#endif

  selectedProfileIndex = loadProfileIndexFromEEPROM();
  activeProfileIndex = selectedProfileIndex;

  if (profileCount == 0)
  {
    // Serial.println("EEPROMに有効なプロファイルが見つかりません。");
    //  プロファイルがないので、colorも設定しない
    return;
  }

  if (selectedProfileIndex >= profileCount)
  {
    // Serial.println("⚠️ EEPROMのプロファイルインデックスが無効です。0にリセットします");
    selectedProfileIndex = 0;
  }

  activeProfileIndex = selectedProfileIndex;
  usbMode = (EEPROM.read(EEPROM_ADDR_USB_MODE) == 1);

  pinMode(BUTTON_PIN, INPUT_PULLUP);
  pinMode(KEY_PIN, INPUT_PULLUP);
  pinMode(MBTN_PIN, INPUT_PULLUP);

  profileSelectStart = millis();

  initLed();
  setDataColor(0, 255, 0);

  ColorProfile *c = wifiProfiles[selectedProfileIndex].color;
  if (!c)
  {
    // Serial.println("⚠️ color が null です！");
    c = &RedProfile; // 安全なデフォルトに退避
    wifiProfiles[selectedProfileIndex].color = c;
  }
  if (usbMode) setLedColor(255, 0, 0);
  else setLedFromProfileIndex(selectedProfileIndex);

  // 10秒間: 短押し=プロファイル切替, 長押し1秒=USB/WiFiモード切替
  // USBモード=赤, WiFiモード=プロファイル色
  while (millis() - profileSelectStart < PROFILE_SELECT_WINDOW)
  {
    if (digitalRead(BUTTON_PIN) == LOW)
    {
      unsigned long pressStart = millis();
      delay(20);
      while (digitalRead(BUTTON_PIN) == LOW && millis() - pressStart < 3000) delay(10);
      unsigned long dur = millis() - pressStart;
      if (dur >= 1000)
      {
        usbMode = !usbMode;
        for (int i = 0; i < 3; i++)
        {
          clearLed(); delay(150);
          if (usbMode) setLedColor(255, 0, 0);
          else setLedFromProfileIndex(selectedProfileIndex);
          delay(150);
        }
      }
      else
      {
        selectedProfileIndex = (selectedProfileIndex + 1) % profileCount;
        if (usbMode) setLedColor(255, 0, 0);
        else setLedFromProfileIndex(selectedProfileIndex);
      }
    }
    delay(20);
  }

  setLedFromProfileIndex(selectedProfileIndex);

  EEPROM.write(EEPROM_ADDR_USB_MODE, usbMode ? 1 : 0);
  EEPROM.commit();
  saveProfileIndexToEEPROM(selectedProfileIndex);
  EEPROM.end();

  activeProfileIndex = selectedProfileIndex;
  gpio_pad_select_gpio((gpio_num_t)BUTTON_PIN);
  gpio_set_direction((gpio_num_t)BUTTON_PIN, GPIO_MODE_DISABLE);
  gpio_install_isr_service(ESP_INTR_FLAG_DEFAULT);
  gpio_set_intr_type((gpio_num_t)KEY_PIN, GPIO_INTR_ANYEDGE);
  gpio_isr_handler_add((gpio_num_t)KEY_PIN, keyISR, NULL);
  // attachInterrupt(digitalPinToInterrupt(KEY_PIN), keyISR, CHANGE);
  if (usbMode) {
    timeSynced = true;
    return;
  }
  connectToWiFi();
  getserverIP();
  Udp.begin(LOCAL_PORT);
  requestMillisSync();
  lastPingMillis = millis();

  Serial.printf("%d", keyRepeatCount);
}

// ==== メインループ ====
void loop()
{
  if (usbMode) {
  // USB中継モード: キー状態をシリアルで送信
  unsigned long nowMillis = millis();
  noInterrupts();
  bool doSend = immediateSendRequested;
  bool sendState = keyState;
  if (doSend) immediateSendRequested = false;
  interrupts();

  if (doSend) {
    currentKeyState = sendState;
    uint8_t cwPkt[10] = {(uint8_t)(sendState ? 0x01 : 0x00), 0x01, 0,0,0,0,0,0,0,1};
    int repeat = sendState ? 3 : 5;
    bool aborted = false;
    for (int i = 0; i < repeat && !aborted; i++) {
      Serial.write(cwPkt, 10);
      if (i < repeat - 1) {
        for (int ms = 0; ms < 5 && !aborted; ms++) {
          delay(1);
          noInterrupts();
          if (immediateSendRequested && keyState != sendState) aborted = true;
          interrupts();
        }
      }
    }
    if (sendState) lastOnMillis = nowMillis;
    setLedColor(0, 0, sendState ? 255 : 20); // キー状態変化時のみLED更新
  }

  // 安全対策: MAX_ON_DURATION超過で強制OFF
  if (currentKeyState && nowMillis - lastOnMillis > MAX_ON_DURATION) {
    currentKeyState = false;
    uint8_t offPkt[10] = {0x00, 0x01, 0,0,0,0,0,0,0,1};
    for (int i = 0; i < 5; i++) { Serial.write(offPkt, 10); if (i < 4) delay(5); }
    setLedColor(0, 0, 20);
  }

  yield(); // delay(1)の代わり: ポーリング遅延を最小化

  return;
  }

  unsigned long nowMicros = micros();
  unsigned long nowMillis = millis();
  byte trxCode = 0x01; // 固定する

  // ==== SYNC再送の強化（初回＋周期） ====

  static bool syncRequested = false;
  static unsigned long lastSyncRetryMillis = 0;

  if (!timeSynced && WiFi.status() == WL_CONNECTED)
  {
    if (!syncRequested || millis() - lastSyncRetryMillis >= 2000)
    {
      requestMillisSync();
      syncRequested = true;
      syncCollecting = true;
      syncSampleIndex = 0;
      lastSyncRetryMillis = millis();
      // Serial.println("SYNC要求を送信しました");
    }
  }

#ifdef M5ATOMLite
  // メカニカルボタン処理
  static bool lastMbtnState = HIGH;
  bool mbtnState = digitalRead(MBTN_PIN);

  if (mbtnState != lastMbtnState)
  {
    lastMbtnState = mbtnState;

    if (mbtnState == LOW)
    {
      keyState = true;
      setDataColor(255, 0, 0);
    }
    else
    {
      keyState = false;
      setDataColor(0, 255, 0);
    }

    immediateSendRequested = true;
    lastKeyChangeMicros = micros();
  }
#endif

  // 即時送信処理
  bool doSend = false;
  bool sendState = false;
  unsigned long isrMicros = 0;
  noInterrupts();
  if (immediateSendRequested)
  {
    doSend = true;
    sendState = keyState;
    isrMicros = lastKeyChangeMicros;
    immediateSendRequested = false;
  }
  interrupts();

  if (doSend && timeSynced)
  {
    fixedDelayLocked = true; // ① 遅延時間の再評価を防止

    currentKeyState = sendState;
    uint64_t opTimeMs = (isrMicros / 1000) + delayOffsetMs + fixedDelayMs;
    int repeat = currentKeyState ? 1 : 3; // OFF時は3回送信
    for (int i = 0; i < repeat; i++)
    {
      sendKeyPacket(currentKeyState, trxCode, opTimeMs); // opTimeMsは固定
#ifdef M5ATOMLite
      yield();
#elif M5ATOMS3Lite
      // delay(1);
      taskYIELD();
#endif
    }

    // lastSendMicros = nowMicros;
    lastOnMillis = currentKeyState ? nowMillis : lastOnMillis;
    lastImmediateSendMillis = millis();
  }

  // ==== SAMPLE_INTERVAL_MSごとのサンプリング処理 ====
  if (timeSynced && millis() - lastSampleTimeMs >= SAMPLE_INTERVAL_MS)
  {
    // 即時送信後は抑制
    //    if (millis() - lastImmediateSendMillis < SAMPLE_SUPPRESS_MS)
    //      return;
    // 即時送信中はサンプリング送信をスキップ
    if (immediateSendRequested)
      return;

    lastSampleTimeMs = millis();
    bool down = (digitalRead(KEY_PIN) == LOW);
    if (sampleIndex == 0)
    {
      sampleStartTimeMs = millis();
      keyBuf = 0; // ← 初期化をここに移動
    }

    sampleTimes[sampleIndex] = millis(); // ← ここに追加！

    if (down)
      keyBuf |= (1 << sampleIndex);
    sampleIndex++;

    if (sampleIndex >= SAMPLE_SEND_AMNT)
    {
      // byte trxCode = (digitalRead(TRX_PIN) == LOW) ? 0x02 : 0x01;
      trxCode = 0x01;
      byte pkt[21]; // 5ヘッダ + 最大16サンプル
      pkt[0] = trxCode;
      pkt[1] = keyBuf;
      pkt[2] = (sampleStartTimeMs >> 16) & 0xFF;
      pkt[3] = (sampleStartTimeMs >> 8) & 0xFF;
      pkt[4] = sampleStartTimeMs & 0xFF;
      for (int i = 0; i < SAMPLE_SEND_AMNT; i++)
      {
        byte delta = (byte)constrain(sampleTimes[i] - sampleStartTimeMs, 0, 255);
        pkt[5 + i] = delta;
      }

      for (int i = 0; i < SAMPLE_SEND_DUP_CNT; i++)
      {
        Udp.beginPacket(serverIP, SERVER_PORT);
        Udp.write(pkt, 5 + SAMPLE_SEND_AMNT);
        Udp.endPacket();
#ifdef M5ATOMLite
        yield();
#elif M5ATOMS3Lite
        // delay(1);
        taskYIELD();
#endif
      }

      sampleIndex = 0;
    }
  }

  // UDP受信処理
  if (WiFi.status() == WL_CONNECTED)
  {
    int size = Udp.parsePacket();
    if (size > 0)
    {
      byte buf[12];
      int len = Udp.read(buf, size);

      if (size == 9 && buf[0] == 0xE1)
      {
        uint64_t serverRecvMs = 0;
        for (int i = 0; i < 8; i++)
          serverRecvMs = (serverRecvMs << 8) | buf[1 + i];

        long offset = (long)serverRecvMs - (long)nowMillis;

        // Serial.printf("🕒 SYNC応答: serverRecvMs=%llu, nowMillis=%lu, delayOffsetMs=%ld\n", serverRecvMs, nowMillis, offset);

        // === サーバー再起動検出 ===
        if (offset < 3000)
        {
          flashRedBeforeRestart();
          ESP.restart();
          return;
        }

        // === 複数回SYNC収集中なら記録 ===
        if (syncCollecting)
        {
          if (syncSampleIndex < SYNC_SAMPLE_COUNT)
          {
            syncOffsets[syncSampleIndex++] = offset;
          }

          if (syncSampleIndex >= SYNC_SAMPLE_COUNT)
          {
            // 最小値を選ぶ
            long minOffset = syncOffsets[0];
            for (int i = 1; i < SYNC_SAMPLE_COUNT; i++)
            {
              if (syncOffsets[i] < minOffset)
                minOffset = syncOffsets[i];
            }

            delayOffsetMs = minOffset;
            timeSynced = true;
            syncCollecting = false;

            // Serial.printf("✅ SYNC完了: 最小 delayOffsetMs=%ld\n", delayOffsetMs);

            // 以下、初期化処理（必要ならここに fixedDelayMs 設定なども）
            fixedDelayLocked = true;
            fixedDelayLockUntil = millis() + 30000;
            blinkInterval = BLINK_INTERVAL_SYNC;
            rttIndex = 0;
            for (int i = 0; i < RTT_HISTORY_SIZE; i++)
              rttHistory[i] = 0;
            sampleIndex = 0;
            keyBuf = 0;
            lastSampleTimeMs = millis();
            lastSyncRequestMillis = millis();
            lastSyncRetryMillis = millis();

            setLedColorFromProfile(&RedProfile);
            delay(100);
            currentKeyState ? setLedColorFromProfile(&GreenProfile) : clearLed();
          }
          else
          {
            requestMillisSync();
            lastSyncRetryMillis = millis();
          }

          return; // まだ収集中なのでここで終了
        }

        delayOffsetMs = offset;

        // 以下、通常のSYNC処理（初期化など）
        timeSynced = true;
        fixedDelayMs = rttToDelayMs(pingRttMs);

        fixedDelayLocked = true;                // 通常はロックして安定運用
        fixedDelayLockUntil = millis() + 30000; // 10秒間ロック維持

        lastPingMillis = nowMillis;
        lastPingSendMillis = nowMillis;
        blinkInterval = BLINK_INTERVAL_SYNC;

        // fixedDelayLocked = false;
        rttIndex = 0;
        for (int i = 0; i < RTT_HISTORY_SIZE; i++)
          rttHistory[i] = 0;

        sampleIndex = 0;
        keyBuf = 0;
        lastSampleTimeMs = nowMillis;
        lastSyncRequestMillis = nowMillis;
        lastSyncRetryMillis = nowMillis;

        // 一瞬赤で点灯
        setLedColorFromProfile(&RedProfile);
        delay(100);

        // currentKeyState に応じて表示
        if (currentKeyState)
        {
          setLedColorFromProfile(&GreenProfile);
        }
        else
        {
          clearLed();
        }
      }

      else if (size == 1 && buf[0] == 0xFE)
      {
        pongReceived = true;

        long rtt = nowMillis - lastPingSendMillis;
        rttHistory[rttIndex] = rtt;
        rttIndex = (rttIndex + 1) % RTT_HISTORY_SIZE;
        pingRttMs = getAverageRtt();

        if (!fixedDelayLocked && !currentKeyState && (millis() - lastOnMillis > 1000))
        {
          fixedDelayMs = rttToDelayMs(pingRttMs);
          fixedDelayLocked = true;
        }

        lastPingMillis = nowMillis;
      }
    }
  }

  // PING送信
  if (timeSynced && nowMillis - lastPingSendMillis > PING_INTERVAL)
  {
    if (!pongReceived)
    {
      if (!currentKeyState)
      { // キー送信中は表示しない
        lostPingCount++;
        // Serial.printf("⚠️ Ping lost! Count: %d (last RTT=%ld ms)\n", lostPingCount, pingRttMs);
      }
      else
      {
        // キー送信中はカウントも止める
        lostPingCount = 0;
      }
    }
    pongReceived = false;

    byte ping = 0xFF;
    Udp.beginPacket(serverIP, SERVER_PORT);
    Udp.write(&ping, 1);
    Udp.endPacket();
    lastPingSendMillis = nowMillis;
  }

  // 遅延時間再計算
  if (timeSynced && nowMillis - lastDelayRecalcMillis >= ReCal_RTT)
  {
    lastDelayRecalcMillis = nowMillis;
    pingRttMs = getAverageRtt();

    if (!fixedDelayLocked)
    {
      fixedDelayMs = rttToDelayMs(pingRttMs);
    }
  }

  // ==== サーバ応答タイムアウト検出 ====
  if (timeSynced && nowMillis - lastPingMillis >= dynamicTimeout)
  {
    timeSynced = false;
    blinkInterval = BLINK_INTERVAL_WIFI;
    fixedDelayLocked = false;
    // Serial.println("サーバとの接続タイムアウト。待機状態に戻ります。");
  }

  // 5秒以上キー押下がない場合は遅延再評価を許可
  if (timeSynced && !currentKeyState && (millis() - lastOnMillis >= 5000))
  {
    fixedDelayLocked = false;
  }

  // LED点滅
  if (nowMillis - lastBlinkMillis >= blinkInterval)
  {
    ledState = !ledState;

    if (ledState)
    {
      setLedColorFromProfile(&GreenProfile);
    }
    else
    {
      clearLed();
    }

    lastBlinkMillis = nowMillis;
  }

  // WiFi切断検出
  if (WiFi.status() != WL_CONNECTED)
  {
    wifiWasDisconnected = true;
    timeSynced = false;
    blinkInterval = BLINK_INTERVAL_WIFI;
    fixedDelayLocked = false;
    static unsigned long lastReconnectMs = 0;
    if (nowMillis - lastReconnectMs >= 30000)
    {
      lastReconnectMs = nowMillis;
      WiFiProfile &rp = wifiProfiles[activeProfileIndex];
      WiFi.begin(rp.ssid, rp.password);
    }
  }

  // ==== SYNC再送の自動化 ====
  if (!timeSynced && WiFi.status() == WL_CONNECTED && millis() - lastSyncRetryMillis >= 5000)
  {
    syncCollecting = true;
    syncSampleIndex = 0;
    requestMillisSync();
    lastSyncRetryMillis = millis();
    // Serial.println("SYNC再送要求を送信しました");
  }

  if (!timeSynced)
    return;

  // ON安全対策
  if (currentKeyState && nowMillis - lastOnMillis > MAX_ON_DURATION)
  {
    currentKeyState = false;
    uint64_t opTimeMs = nowMillis + delayOffsetMs + fixedDelayMs;
    sendKeyPacket(false, trxCode, opTimeMs);
  }

  if (fixedDelayLocked && millis() > fixedDelayLockUntil)
  {
    fixedDelayLocked = false;
  }

  delay(1);
}

// ==== WiFi接続 ====
bool connectToWiFi()
{

  WiFiProfile profile = wifiProfiles[activeProfileIndex]; // ← 選択済みのプロファイルだけ使う！
  if (!profile.isApMode && profile.hostname && strlen(profile.hostname) > 0)
  {
    WiFi.setHostname(profile.hostname);
  }
  if (profile.hasIP)
  {
    WiFi.config(profile.localIP, profile.gateway, profile.subnet);
  }
  esp_wifi_set_ps(WIFI_PS_NONE);
  WiFi.begin(profile.ssid, profile.password);

  if (profile.isApMode)
  {
    esp_wifi_set_max_tx_power(84);
  }

  for (int t = 0; t < 60; t++)
  {
    if (WiFi.status() == WL_CONNECTED)
    {
      serverIP = profile.serverIP;
      int rssi = WiFi.RSSI();
      dynamicTimeout = DEFAULT_TIMEOUT;

      // Serial.printf("✅ 接続成功: SSID=%s, IP=%s\n", profile.ssid, WiFi.localIP().toString().c_str());
      // Serial.printf("✅ 接続成功: SSID=%s", profile.ssid);
      return true;
    }
    if (WiFi.status() == WL_NO_SSID_AVAIL)
    {
      // Serial.println("⏳ SSIDが見つかりません。再試行中...");
    }
    delay(500);
  }

  // Serial.printf("❌ 接続失敗: SSID=%s, IP=%s, status=%d\n", profile.ssid, WiFi.localIP().toString().c_str(), WiFi.status());

  return false;
}

// ==== SYNC要求（時刻補正専用） ====
void requestMillisSync()
{
  lastSyncRequestMillis = millis();
  lastPingSendMillis = lastSyncRequestMillis;
  byte pkt[5] = {
      0xE0,
      (byte)((lastSyncRequestMillis >> 24) & 0xFF),
      (byte)((lastSyncRequestMillis >> 16) & 0xFF),
      (byte)((lastSyncRequestMillis >> 8) & 0xFF),
      (byte)(lastSyncRequestMillis & 0xFF)};
  Udp.beginPacket(serverIP, SERVER_PORT);
  Udp.write(pkt, 5);
  Udp.endPacket();
}

// ==== キーパケット送信 ====
void sendKeyPacket(bool isOn, byte trxSel, uint64_t opTimeMs)
{
  byte pkt[10];
  pkt[0] = isOn ? 0x01 : 0x00;
  pkt[1] = trxSel;
  for (int i = 0; i < 8; i++)
    pkt[2 + i] = (opTimeMs >> (56 - i * 8)) & 0xFF;

  for (int i = 0; i < keyRepeatCount; i++)
  {
    if (!Udp.beginPacket(serverIP, SERVER_PORT))
      return;
    Udp.write(pkt, 10);
    Udp.endPacket();
#ifdef M5ATOMLite
    yield();
#elif M5ATOMS3Lite
    // delay(1);
    taskYIELD();
#endif
  }

  if (isOn)
  {
    setLedColorFromProfile(&RedProfile);
  }
  else
  {
    clearLed();
  }
}

void flashRedBeforeRestart()
{
  for (int i = 0; i < 3; i++)

    setLedColorFromProfile(&RedProfile);
    delay(100);
    clearLed();
    delay(100);
  }
}

bool loadWiFiProfilesFromSD()
{
  File file = SD.open(SD_Filename);
  if (!file)
    return false;

  WiFiProfile temp;
  ColorProfile *color = &cyanProfile;
  profileCount = 0;
  bool hasData = false;

  while (file.available())
  {
    String line = file.readStringUntil('\n');
    line.trim();

    if (line.length() == 0 || line.startsWith("#"))
      continue;

    if (line.startsWith("[Profile]"))
    {
      if (hasData && temp.ssid && strlen(temp.ssid) > 0 && temp.password && strlen(temp.password) > 0 && profileCount < MAX_WIFI_PROFILES)
      {
        wifiProfiles[profileCount] = temp;
        wifiProfiles[profileCount].color = color;
        /*Serial.printf("✅ プロファイル[%d] 読み込み: SSID=%s, MODE=%s, COLOR=%s\n",
                      profileCount,
                      temp.ssid,
                      temp.isApMode ? "AP" : "STA",
                      color->name);
                      */
        profileCount++;
      }
      else if (hasData)
      {
        // Serial.println("⚠️ 無効なプロファイル（SSIDまたはPASSなし）をスキップしました");
      }

      freeWiFiProfileStrings(temp);
      temp = WiFiProfile();
      color = &cyanProfile;
      hasData = false;
    }
    else if (line.startsWith("SSID="))
    {
      temp.ssid = strdup(line.substring(5).c_str());
      hasData = true;
    }
    else if (line.startsWith("PASS="))
    {
      temp.password = strdup(line.substring(5).c_str());
      hasData = true;
    }
    else if (line.startsWith("IP="))
      temp.localIP.fromString(line.substring(3));
    else if (line.startsWith("GW="))
      temp.gateway.fromString(line.substring(3));
    else if (line.startsWith("MASK="))
      temp.subnet.fromString(line.substring(5));
    else if (line.startsWith("SERVER="))
      temp.serverIP.fromString(line.substring(7));
    else if (line.startsWith("MODE="))
      temp.isApMode = (line.substring(5) == "AP");
    else if (line.startsWith("HASIP="))
      temp.hasIP = (line.substring(6) == "IP");
    else if (line.startsWith("COLOR="))
    {
      String c = line.substring(6);
      if (c == "Magenta")
        color = &MagentaProfile;
      else if (c == "Yellow")
        color = &yellowProfile;
      else if (c == "Cyan")
        color = &cyanProfile;
      else if (c == "Red")
        color = &RedProfile;
      else if (c == "Blue")
        color = &BlueProfile;
      else if (c == "Green")
        color = &GreenProfile;
      else if (c == "White")
        color = &WhiteProfile;
    }
    else if (line.startsWith("HOSTNAME="))
    {
      temp.hostname = strdup(line.substring(9).c_str());
    }
    else if (line.startsWith("SERVERHOSTNAME="))
    {
      temp.serverhostname = strdup(line.substring(15).c_str());
    }
    else if (line.startsWith("[Settings]"))
    {
      while (file.available())
      {
        String sline = file.readStringUntil('\n');
        sline.trim();
        if (sline.length() == 0 || sline.startsWith("#"))
          continue;
        if (sline.startsWith("["))
        {
          // 次のセクションに入ったら戻す
          line = sline;
          break;
        }

        if (sline.startsWith("KEY_REPEAT="))
          keyRepeatCount = constrain(sline.substring(11).toInt(), 1, 50);
        else if (sline.startsWith("SAMPLE_DUP="))
          SAMPLE_SEND_DUP_CNT = constrain(sline.substring(11).toInt(), 1, 10);
        else if (sline.startsWith("SAMPLE_INTERVAL="))
          SAMPLE_INTERVAL_MS = constrain(sline.substring(16).toInt(), 1, 100);
        else if (sline.startsWith("SAMPLE_AMOUNT="))
          SAMPLE_SEND_AMNT = constrain(sline.substring(14).toInt(), 1, 16);
        else if (sline.startsWith("DEBOUNCE_US="))
          DEBOUNCE_US = constrain(sline.substring(12).toInt(), 100, 50000);
      }
    }
  }

  // 最後のプロファイルを保存
  if (hasData && temp.ssid && strlen(temp.ssid) > 0 && temp.password && strlen(temp.password) > 0 && profileCount < MAX_WIFI_PROFILES)
  {
    wifiProfiles[profileCount] = temp;
    wifiProfiles[profileCount].color = color;
    /*Serial.printf("✅ プロファイル[%d] 読み込み: SSID=%s, MODE=%s, COLOR=%s\n",
                  profileCount,
                  temp.ssid,
                  temp.isApMode ? "AP" : "STA",
                  color->name);
                  */
    profileCount++;
  }
  else if (hasData)
  {
    // Serial.println("⚠️ 最後のプロファイルが無効（SSIDまたはPASSなし）だったためスキップしました");
  }

  file.close();
  // Serial.printf("📦 SDから読み込んだプロファイル数: %d\n", profileCount);
  return profileCount > 0;
}

void saveWiFiProfileToEEPROM(const WiFiProfile &src, int index)
{
  if (!src.ssid || strlen(src.ssid) == 0 || !src.password || strlen(src.password) == 0)
  {
    // Serial.printf("❌ saveWiFiProfileToEEPROM: 無効なプロファイル（index=%d）をスキップします\n", index);
    return;
  }

  WiFiProfileEEPROM dst;
  memset(&dst, 0, sizeof(dst));
  dst.magic = PROFILE_MAGIC;
  strncpy(dst.ssid, src.ssid, MAX_SSID_LEN);
  strncpy(dst.password, src.password, MAX_PASS_LEN);
  if (src.hostname) strncpy(dst.hostname, src.hostname, MAX_HOSTNAME_LEN);
  if (src.serverhostname) strncpy(dst.serverhostname, src.serverhostname, MAX_HOSTNAME_LEN);

  for (int i = 0; i < 4; i++)
  {
    dst.localIP[i] = src.localIP[i];
    dst.gateway[i] = src.gateway[i];
    dst.subnet[i] = src.subnet[i];
    dst.serverIP[i] = src.serverIP[i];
  }

  dst.isApMode = src.isApMode;
  dst.hasIP = src.hasIP;
  if (src.color == &cyanProfile)
    dst.colorId = 0;
  else if (src.color == &MagentaProfile)
    dst.colorId = 1;
  else if (src.color == &yellowProfile)
    dst.colorId = 2;
  else if (src.color == &RedProfile)
    dst.colorId = 3;
  else if (src.color == &BlueProfile)
    dst.colorId = 4;
  else if (src.color == &GreenProfile)
    dst.colorId = 5;
  else if (src.color == &WhiteProfile)
    dst.colorId = 6;
  else
    dst.colorId = 0;

  int addr = EEPROM_ADDR_PROFILE_BASE + index * sizeof(WiFiProfileEEPROM);
  // Serial.printf("💾 書き込み index=%d, addr=%d, SSID=%s\n", index, addr, src.ssid);
  EEPROM.put(addr, dst);
  EEPROM.commit();
}

bool loadWiFiProfileFromEEPROM(WiFiProfile &dst, int index)
{
  WiFiProfileEEPROM raw;
  int addr = EEPROM_ADDR_PROFILE_BASE + index * sizeof(WiFiProfileEEPROM);
  EEPROM.get(addr, raw);

  // Serial.printf("📖 読み込み index=%d, addr=%d, magic=0x%02X\n", index, addr, raw.magic);

  if (raw.magic != PROFILE_MAGIC)
  {
    // Serial.printf("❌ 無効なプロファイル index=%d（magic=0x%02X）\n", index, raw.magic);
    return false; // ← if の中に入れる！
  }

  freeWiFiProfileStrings(dst);
  dst.ssid = strdup(raw.ssid);
  dst.password = strdup(raw.password);
  dst.localIP = IPAddress(raw.localIP);
  dst.gateway = IPAddress(raw.gateway);
  dst.subnet = IPAddress(raw.subnet);
  dst.serverIP = IPAddress(raw.serverIP);
  dst.isApMode = raw.isApMode;
  dst.hasIP = raw.hasIP;
  dst.hostname = strdup(raw.hostname);
  dst.serverhostname = strdup(raw.serverhostname);

  switch (raw.colorId)
  {
  case 0:
    dst.color = &cyanProfile;
    break;
  case 1:
    dst.color = &MagentaProfile;
    break;
  case 2:
    dst.color = &yellowProfile;
    break;
  case 3:
    dst.color = &RedProfile;
    break;
  case 4:
    dst.color = &BlueProfile;
    break;
  case 5:
    dst.color = &GreenProfile;
    break;
  case 6:
    dst.color = &WhiteProfile;
    break;
  default:
    // Serial.printf("⚠️ EEPROMの colorId=%d は不正です。Red にフォールバックします\n", raw.colorId);
    dst.color = &RedProfile;
    break;
  }

  return true;
}

void clearEEPROM()
{
  for (int i = 0; i < EEPROM_SIZE; i++)
    EEPROM.write(i, 0xFF);
  EEPROM.commit();
  delay(100);
}

bool restoreProfilesFromEEPROM()
{
  profileCount = 0;
  for (int i = 0; i < MAX_WIFI_PROFILES; i++)
  {
    if (loadWiFiProfileFromEEPROM(wifiProfiles[i], i))
    {
      profileCount++;
      // Serial.printf("✅ EEPROMからプロファイル[%d] を復元: SSID=%s\n", i, wifiProfiles[i].ssid);
    }
  }
  // Serial.printf("📊 復元されたプロファイル数: %d\n", profileCount);
  return profileCount > 0;
}

void getserverIP()
{
  WiFiProfile &profile = wifiProfiles[activeProfileIndex];

  if (!profile.hasIP && profile.serverhostname != nullptr && strlen(profile.serverhostname) > 0)
  {
    if (!MDNS.begin(profile.hostname))
    {

      // Serial.println("❌ クライアント側 mDNS初期化失敗");
      return;
    }
    // Serial.println("✅ クライアント側 mDNS初期化成功");
    // Serial.print("📡 クライアントIP: ");
    // Serial.println(WiFi.localIP());
    String hostname = profile.serverhostname;
    delay(1000);
    if (!hostname.endsWith(".local"))
    {
      hostname += ".local";
    }

    // Serial.print("🔍 mDNSクエリ中: ");
    // Serial.println(hostname);

    String queryName = hostname;
    queryName.replace(".local", "");

    IPAddress resolved;
    for (int i = 0; i < HOSTNAME_RESTIME; i++)
    {
      resolved = MDNS.queryHost(queryName);
      if (resolved != INADDR_NONE)
        break;
      // Serial.println("⏳ mDNSリトライ中...");
      delay(1000);
    }
    if (resolved == INADDR_NONE)
    {
      // Serial.println("❌ mDNSクエリ失敗（タイムアウト）");
      return;
    }

    serverIP = resolved;
    // Serial.print("✅ mDNSで解決: ");
    // Serial.println(serverIP);
  }
  else
  {
    serverIP = profile.serverIP;
  }
}

void saveSystemSettingsToEEPROM(const SystemSettingsEEPROM &settings)
{
  int addr = EEPROM_ADDR_SETTINGS;
  const byte *ptr = (const byte *)&settings;
  for (size_t i = 0; i < sizeof(SystemSettingsEEPROM); i++)
  {
    EEPROM.write(addr + i, ptr[i]);
  }
  EEPROM.commit();
}

bool loadSystemSettingsFromEEPROM(SystemSettingsEEPROM &settings)
{
  int addr = EEPROM_ADDR_SETTINGS;
  byte *ptr = (byte *)&settings;
  for (size_t i = 0; i < sizeof(SystemSettingsEEPROM); i++)
  {
    ptr[i] = EEPROM.read(addr + i);
  }
  return settings.magic == 0xA5;
}
